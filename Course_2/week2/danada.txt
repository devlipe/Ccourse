void shuffle_cards(cards deck[], int amount)
{
    srand(time(NULL));
    int changer = 0;
    cards temp; 

    for (int i = 0; i < amount; i++)
    {
        changer = rand()% amount;

        temp = deck[i];
        deck[i] = deck[changer];
        deck[changer] = temp;
    }
}
void print_cards(cards deck[], int amount)
{
    for(int i = 0; i < amount; i++)
    {
        printf("%s of %s\n",print_pips(deck[i].pips), print_suit(deck[i].suit));
    }
}
int probability (cards deck[], int full_house, int four_kind, int three_kind, int two_pair, int pair, int no_pair, int amount)
{
    
    for (int p = 0; p < 1000000; p++)
    {
        int count = 0, t = 100, change = 0;
        shuffle_cards(deck, amount);
 
        for (int i = 0; i < 7; i++)
        {
            for (int j = i + 1; j < 7; j++)
            {
                if(deck[i].pips == deck[j].pips)
                {
                    if (t != deck[i].pips)
                    {
                        change +=1;
                        count += 1;
                    }
                    
                    t = deck[i].pips;
                    count++;
                    break;
                }
            }
        }
        if (count == 5)
        {
            full_house+=1;
        }
        else if (count == 4)
        {
            if (change == 1)
            {
                four_kind+=1;
            }
            else if (change == 2)
            {
                two_pair+=1;
            }
        }
        else if (count == 3)
        {
            three_kind+=1;
        }
        else if (count == 2)
        {
            pair+=1;
        }
        else if (count == 0)
        {
        no_pair+=1;
        }
    }  
    return (full_house,four_kind, three_kind, two_pair, pair, no_pair, amount);
}
